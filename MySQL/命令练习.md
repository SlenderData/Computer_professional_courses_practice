## 题目

```mysql
select *
from 学生表;

select 姓名, 性别, 出生日期
from 学生表
where 性别 = "F";

select *
from 学生表
where year(curdate()) - year(出生日期) > 23;

select *
from 学生表
where 专业 = "会计";

select 学号
from 成绩表
where 课程名称 = "高数"
  and 成绩 between 80 and 90;

select 学生表.学号, 姓名, 课程名称, 成绩
from 学生表,
     成绩表
where 课程编号 = "14"
  and 学生表.学号 = 成绩表.学号;

select 学生表.学号, 姓名, 课程名称, 成绩
from 学生表,
     成绩表
where 学生表.学号 = "992101"
  and 学生表.学号 = 成绩表.学号;



select *
from 成绩表
order by 成绩 desc;

select *
from 成绩表
order by 学号 asc, 成绩 desc;

select 学号, sum(成绩) as 总分
from 成绩表
where 学号 = "992105";

select 课程名称, sum(成绩) as 总分, avg(成绩) as 平均分
from 成绩表
where 课程名称 = "高数";

select 姓名, count(课程编号) as 选修课程总数
from 成绩表,
     学生表
where 成绩表.学号 = 学生表.学号
group by 姓名;

select count(学号) as 男同学总数
from 学生表
where 性别 = "F";

select 学号
from 成绩表
where 成绩 > 80
  and 课程名称 = "体育";

select 课程名称, sum(成绩) as 总分, avg(成绩) as 平均分
from 成绩表
group by 课程名称;

select 课程名称, count(学号) as 选修人数
from 成绩表
group by 课程名称;

select 课程名称, avg(成绩) as 平均分
from 成绩表
group by 课程名称
having avg(成绩) > 80;
```
